Java Switch Statement Validator

import re

def validate(filename):
    code = open(filename).read()
    code = re.sub(r'//.*', '', code)
    code = re.sub(r'\s+', ' ', code).strip()
    
    if code.count('{') != code.count('}'):
        print("Syntax error: Missing braces")
        return
    
    if code.count('(') != code.count(')'):
        print("Syntax error: Missing parenthesis")
        return
    
    expr = r'\([^()]+\)'
    stmt = r'[^:{};]+;'
    case_block = rf'case\s+[^:]+:\s*({stmt}\s*)*(break\s*;)?'
    default_block = rf'default\s*:\s*({stmt}\s*)+'
    switch_pattern = rf'''^switch\s*{expr}\s*\{{\s*({case_block}\s*)+{default_block}\s*\}}$'''
    
    if re.match(switch_pattern, code, re.VERBOSE):
        print("Valid Java switch statement")
    else:
        print("Invalid Java switch statement")

validate('switch.txt')

def separate(expr):
    vals, ops = [], []
    temp = ""
    for c in expr:
        if c.isalnum():
            temp += c
        else:
            if temp:
                vals.append(temp)
                temp = ""
            ops.append(c)
    if temp:
        vals.append(temp)
    return vals, ops

expr = input("Enter expression: ").replace(" ", "")

    var, expr = expr.split("=")
else:
    var = None

if len(vals) != len(ops) + 1:
    print("Invalid expression")
else:
    print("Operands:", vals)
    print("Operators:", ops)
    
    print("\nTriplets:")
    t = 1
    last = ""
    for i, op in enumerate(ops):
        if i == 0:
            last = f"t{t}"
            print(f"{last} = {vals[0]} {op} {vals[1]}")
        else:
            print(f"t{t} = {last} {op} {vals[i+1]}")
            last = f"t{t}"
        t += 1

    t = 1
    last = ""

    tac_rows = []
    
    for i in range(len(o)):
        t = "T" + str(i + 1)
        if i == 0:
            arg1, op, arg2 = v[0], o[0], v[1]
        else:
            arg1, op, arg2 = last, o[i], v[i + 1]
        tac_rows.append((t, arg1, op, arg2))
        last = t
    
    # Print TAC table
    print("Temp\tArg1\tOp\tArg2")
    for row in tac_rows:

table = {}

    line = input().strip()
    if line == "end":
        break
    
    line = line.replace(" ", "").rstrip(";")
    
    # -------- DECLARATION --------
    if line.startswith(("int", "float", "char")):
        if line.startswith("int"):
            var_type = "int"
            vars_part = line[3:]
        elif line.startswith("float"):
            var_type = "float"
            vars_part = line[5:]
        else:
            var_type = "char"
            vars_part = line[4:]
        
        variables = vars_part.split(",")
        for var in variables:
            table[var] = {
                "type": var_type,
                "addr": random.randint(1000, 9999),
                "defined": 1,
                "used": 0
            }
    else:
        parts = line.split("=")
        if len(parts) >= 2:
            left = parts[0]
            if left not in table:
                table[left] = {
                    "type": "unknown",
                    "addr": random.randint(1000, 9999),
                    "defined": 0,
                    "used": 1
                }
        
        for var in table:
            if var in line:
                table[var]["used"] = 1

print("\nSymbol Table:")
print("Name\tType\tAddr\tDefined\tUsed")
for name, info in table.items():
    print(f"{name}\t{info['type']}\t{info['addr']}\t{info['defined']}\t{info['used']}")

Three Address to Mnemonic Code Converter

def convert_to_mnemonic(line):
    line = line.replace(" ", "")
    if "=" not in line:
        return "Invalid"
    
    left, right = line.split("=")
    
    # Case 1: Simple assignment → a = b
    if right.isalnum():
        return f"MOV {left}, {right}"
    
    # Case 2: Binary operation → a = b + c
    for op in ['+', '-', '*', '/']:
        if op in right:
            op1, op2 = right.split(op)
            if op == '+':
                return f"MOV R1, {op1}\nADD R1, {op2}\nMOV {left}, R1"
            elif op == '-':
                return f"MOV R1, {op1}\nSUB R1, {op2}\nMOV {left}, R1"
            elif op == '*':
                return f"MOV R1, {op1}\nMUL R1, {op2}\nMOV {left}, R1"
            elif op == '/':
                return f"MOV R1, {op1}\nDIV R1, {op2}\nMOV {left}, R1"
    
    return "Invalid"

print("Enter 3-address code (type 'end' to stop):")
while True:
    line = input()
    if line == "end":
        break
    print("\nMnemonic Code:")
    print(convert_to_mnemonic(line))

Java Loop Validator

import re

def validate_loops(filename):
    code = open(filename).read()
    code = re.sub(r'//.*', '', code)
    code = re.sub(r'\s+', ' ', code).strip()
    
    while_pattern = r'while\s*\([^()]+\)\s*\{[^{}]*\}'
    for_pattern = r'for\s*\([^;]+;[^;]+;[^()]+\)\s*\{[^{}]*\}'
    do_while_pattern = r'do\s*\{[^{}]*\}\s*while\s*\([^()]+\)\s*;'
    
    if code.count('{') != code.count('}'):
        print("Syntax error: Missing braces")
        return
    
    found = False
    
    for match in re.findall(while_pattern, code):
        print("Valid Java while loop")
        found = True
    
    for match in re.findall(for_pattern, code):
        print("Valid Java for loop")
        found = True
    
    for match in re.findall(do_while_pattern, code):
        print("Valid Java do-while loop")
        found = True
    
    if not found:
        print("No valid Java loops found")

validate_loops('loop.txt')

Finite Automata String Validator

import numpy as np

transition = {}

while True:
    states = input("Enter states (and type 'done' to finish): ")
    if states.lower() == 'done':
        break
    transition[states] = {}

for outer_key in transition:
    print(f"\nNow entering details for '{outer_key}':")
    while True:
        inner_key = input(f"Enter symbol for '{outer_key}' (type 'done' to finish): ")
        if inner_key.lower() == 'done':
            break
        inner_value = input(f"Enter the state for '{inner_key}': ")
        transition[outer_key][inner_key] = inner_value

print(transition)

start_state = input("Enter the Start state:")
final_state = input("Enter the final state:")
strr = input("Enter the string")

def traverse(start_state, final_state, strr, transition):
    current_state = start_state
    for symbol in strr:
        if symbol not in transition[current_state]:
            print(f"{symbol} is not valid for {current_state} state")
            return False
        current_state = transition[current_state][symbol]
        print(f"On symbol '{symbol}' → moved to '{current_state}'")
    return current_state == final_state

if traverse(start_state, final_state, strr, transition):
    print("\nString accepted: final state reached ")
else:
    print("\nString rejected: final state NOT reached ")
