Basic Blockchain Implementation

import hashlib
import time

class Block:
    def __init__(self, index, data, previous_hash):
        self.index = index
        self.timestamp = time.time()
        self.data = data
        self.previous_hash = previous_hash
        self.hash = self.calculate_hash()
    
    def calculate_hash(self):
        value = str(self.index) + str(self.timestamp) + str(self.data) + str(self.previous_hash)
        return hashlib.sha256(value.encode()).hexdigest()

Proof of Work Algorithm

def proof_of_work(index, data, previous_hash, difficulty):
    nonce = 0
    while True:
        guess = f"{index}{data}{previous_hash}{nonce}"
        guess_hash = hashlib.sha256(guess.encode()).hexdigest()
        if guess_hash[:difficulty] == '0' * difficulty:
            return nonce, guess_hash
        nonce += 1

nonce, hash_val = proof_of_work(1, "Test Block", "0", 4)
print("Nonce Found:", nonce)
print("Hash with Proof of Work:", hash_val)

Flask Blockchain API

import hashlib
import time
from flask import Flask, jsonify

class Block:
    def __init__(self, index, data, previous_hash):
        self.index = index
        self.timestamp = time.time()
        self.data = data
        self.previous_hash = previous_hash
        self.hash = self.calculate_hash()
    
    def calculate_hash(self):
        value = str(self.index) + str(self.timestamp) + str(self.data) + str(self.previous_hash)
        return hashlib.sha256(value.encode()).hexdigest()

app = Flask(__name__)
blockchain = []

@app.route('/create_genesis', methods=['GET'])
def create_genesis():
    if len(blockchain) == 0:
        genesis = Block(0, "Genesis Block", "0")
        blockchain.append(genesis)
        return jsonify({
            "message": "Genesis block created",
            "hash": genesis.hash
        })
    return jsonify({"message": "Genesis block already exists"})

@app.route('/chain', methods=['GET'])
def get_chain():
    chain_data = []
    for block in blockchain:
        chain_data.append({
            "index": block.index,
            "timestamp": block.timestamp,
            "data": block.data,
            "previous_hash": block.previous_hash,
            "hash": block.hash
        })
    return jsonify(chain_data)

if __name__ == '__main__':
    app.run(debug=True)

        uint total = 0;
        for(uint i = 0; i < array.length; i++) {
            total += array[i];
        }

Solidity - Palindrome Checker

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Palindrome {
    function isPalindrome(uint num) public pure returns (bool) {
        uint original = num;
        uint reverse = 0;
        
        while(num > 0) {
            uint digit = num % 10;
            reverse = reverse * 10 + digit;
            num = num / 10;
        }
        
        return original == reverse;
    }
}

Solidity - Fibonacci Contract

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Fibonacci {
    function fib(uint n) public pure returns (uint) {
        if (n == 0) return 0;
        if (n == 1) return 1;
        
        uint a = 0;
        uint b = 1;
        uint c;
        
        for(uint i = 2; i <= n; i++) {
            c = a + b;
            a = b;
            b = c;
        }
        
        return c;
    }
}

Solidity - Calculator Contract

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Calculator {
    function calc(uint a, uint b) public pure returns (uint, int, uint, uint, uint) {
        uint sum = a + b;
        int sub = int(a) - int(b);
        uint mul = a * b;
        uint divi = 0;
        uint mod = 0;
        
        if (b != 0) {
            divi = a / b;
            mod = a % b;
        }
        
        return (sum, sub, mul, divi, mod);
    }
}

Solidity - Struct & Mapping Contract

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract MyContract {
    uint256 public peopleCount = 0;
    
    // Mapping from uint to Person struct
    mapping(uint => Person) public people;
    
    // Define a struct for Person
    struct Person {
        uint id;
        string firstName;
        string lastName;
    }
    
    // Function to add a new person
    function addPerson(string memory firstName, string memory lastName) public {
        incrementCount(); // Increase the count
        people[peopleCount] = Person(peopleCount, firstName, lastName); // Store person
    }
    
    // Internal function to increment the count
    function incrementCount() internal {
        peopleCount += 1;
    }
}
